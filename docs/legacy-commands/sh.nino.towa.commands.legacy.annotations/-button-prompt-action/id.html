id

The ID of the button. This HAS to be the same as the custom_id property, wildcards are also available if it uses dynamic data.

Example Button Prompt

@Command("some-prompt", "Represents a button prompt.")
class MyCommand: AbstractCommand() {
    override suspend fun execute(ctx: TowaCommandContext, @Arg("uwu") uwu: String) {
        ctx.reply("Do you accept the Terms of Service of $uwu?") {
            button("yes", PRIMARY)
            button("no", DANGER)
        }
    }

    @ButtonPromptAction("yes")
    suspend fun onYes(ctx: TowaButtonPromptContext) {
       ctx.reply("ur cool :D")
    }

    @ButtonPromptAction("no")
    suspend fun onNo(ctx: TowaButtonPromptContext) {
       ctx.reply("ur dumb :(")
    }
}
